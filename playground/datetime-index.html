DatetimeIndex & date_range

Timezone-aware datetime index — a TypeScript port of pandas.DatetimeIndex and pd.date_range.

1. Constructing a DatetimeIndex

A DatetimeIndex can be built from ISO strings, Date objects, or millisecond timestamps. 

import { DatetimeIndex } from "tsb";

// From ISO strings
const idx = new DatetimeIndex([
  "2024-01-15",
  "2024-06-30",
  "2024-12-31",
]);

console.log(idx.size);   // 3
console.log(idx.values[0].toISOString());
// "2024-01-15T00:00:00.000Z"

// From ms timestamps
const ms = DatetimeIndex.fromMs([
  Date.UTC(2024, 0, 1),
  Date.UTC(2024, 6, 1),
]);

2. Generating ranges with date_range

date_range(start, options) creates evenly-spaced datetime sequences — just like pd.date_range. 

import { date_range } from "tsb";

// Daily range, 5 periods
const daily = date_range("2024-01-01", {
  periods: 5,
  freq: "D",
});
console.log(daily.strftime("%Y-%m-%d"));
// ["2024-01-01", ..., "2024-01-05"]

// Monthly starts for Q1
const monthly = date_range("2024-01-01", {
  end: "2024-03-01",
  freq: "MS",
});
console.log(monthly.month); // [1, 2, 3]

// Hourly range
const hourly = date_range("2024-01-01T00:00:00Z", {
  periods: 6,
  freq: "H",
});
console.log(hourly.hour); // [0,1,2,3,4,5]

3. Component accessors

Access datetime components element-wise — mirrors pandas' vectorized .year, .month, .dayofweek, etc. 

import { date_range } from "tsb";

const rng = date_range("2024-01-01", {
  periods: 4,
  freq: "QS",  // quarter starts
});

console.log(rng.year);    // [2024,2024,2024,2024]
console.log(rng.month);   // [1, 4, 7, 10]
console.log(rng.quarter); // [1, 2, 3, 4]
console.log(rng.is_quarter_start); // [true,true,true,true]
console.log(rng.is_leap_year);     // [true,true,true,true]
console.log(rng.dayofweek); // 0=Mon … 6=Sun

4. Formatting with strftime

Format each datetime using strftime-style directives, matching pandas' DatetimeIndex.strftime. 

import { date_range } from "tsb";

const rng = date_range("2024-01-01", {
  periods: 3,
  freq: "MS",
});

// ISO date
console.log(rng.strftime("%Y-%m-%d"));

// Full weekday + month
console.log(rng.strftime("%A, %B %d %Y"));

// Compact
console.log(rng.strftime("%y/%m/%d %H:%M"));

5. Floor, Ceil & Round

Snap datetimes to the nearest frequency boundary, matching pandas' .floor(), .ceil(), and .round(). 

import { DatetimeIndex } from "tsb";

const idx = new DatetimeIndex([
  "2024-01-15T10:37:25.123Z",
  "2024-06-30T23:50:00.000Z",
]);

console.log(idx.floor("H").strftime("%H:%M:%S"));
// ["10:00:00", "23:00:00"]

console.log(idx.ceil("H").strftime("%H:%M:%S"));
// ["11:00:00", "00:00:00"]

console.log(idx.round("H").strftime("%H:%M:%S"));
// ["11:00:00", "00:00:00"]

// Floor to day
console.log(idx.floor("D").strftime("%Y-%m-%d %H:%M"));

6. Timezone operations

DatetimeTZDtype represents timezone-aware datetime dtypes (datetime64[ns, tz]). Use tz_localize and tz_convert to work with timezones. 

import { DatetimeIndex, DatetimeTZDtype } from "tsb";

// Create tz-aware dtype
const dtype = new DatetimeTZDtype("America/New_York");
console.log(dtype.toString());
// "datetime64[ns, America/New_York]"

// Localize naive index to UTC
const naive = new DatetimeIndex(["2024-06-15T12:00:00Z"]);
const utcIdx = naive.tz_localize("UTC");
console.log(utcIdx.tz);    // "UTC"
console.log(utcIdx.dtype.toString());

// Convert timezone (only changes annotation, not UTC value)
const nyIdx = utcIdx.tz_convert("America/New_York");
console.log(nyIdx.tz);     // "America/New_York"
// UTC timestamps are unchanged:
console.log(nyIdx.asi8[0] === utcIdx.asi8[0]); // true

7. Shifting datetimes

Shift all datetimes by n periods of a given frequency — supports both fixed (D, H, T, S) and calendar (MS, ME, QS, YS) frequencies. 

import { date_range } from "tsb";

const rng = date_range("2024-01-01", {
  periods: 3, freq: "MS",
});
console.log(rng.strftime("%Y-%m-%d"));
// ["2024-01-01", "2024-02-01", "2024-03-01"]

// Shift forward by 2 months
const shifted = rng.shift(2, "MS");
console.log(shifted.strftime("%Y-%m-%d"));
// ["2024-03-01", "2024-04-01", "2024-05-01"]

// Shift backward by 5 days
const back = rng.shift(-5, "D");
console.log(back.strftime("%Y-%m-%d"));
