Code example

import { Series, DataFrame, astype, dataFrameAstype } from "tsb";

// Cast strings to integers
const s = new Series({ data: ["1", "2", "3"] });
const ints = astype(s, "int64");
// ints.values → [1, 2, 3]
// ints.dtype.name → "int64"

// Ignore unconvertible values instead of throwing
const s2 = new Series({ data: ["a", "2", "3"] });
const safe = astype(s2, "int64", { errors: "ignore" });
// safe.values → [null, 2, 3]

// Per-column cast on a DataFrame
const df = DataFrame.fromColumns({
  price: ["9.99", "14.99", "4.50"],
  inStock: [1, 0, 1],
});
const cast = dataFrameAstype(df, { price: "float64", inStock: "bool" });
// cast.col("price").values → [9.99, 14.99, 4.5]
// cast.col("inStock").values → [true, false, true]
